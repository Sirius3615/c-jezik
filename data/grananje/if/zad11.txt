#include <stdio.h>
#include <math.h>
int x;
float fx;
void main()
{
	printf("\n Upisite broj x: ");
	scanf("%d",&x);
	if (x<0) fx=pow(x,2);
	if (x==0) fx=0;
	if (x>0) fx=sqrt(x);
	printf(" f(x)=%.2f",fx);
}
